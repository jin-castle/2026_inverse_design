Standard setup ALL patterns need:
```python
import meep as mp
import meep.adjoint as mpa
import numpy as np
try:
    import autograd.numpy as npa
except ImportError:
    npa = np
import matplotlib
matplotlib.use("Agg")
import matplotlib.pyplot as plt
from pathlib import Path
import json, time, os, sys

silicon = mp.Medium(index=3.48)
oxide   = mp.Medium(index=1.44)
air     = mp.Medium(index=1.0)
SILICON = silicon; OXIDE = oxide  # aliases

wavelength = 1.55
fcen       = 1.0 / wavelength
fwidth     = 0.2 * fcen
resolution = 20
dpml       = 1.0
wg_width   = 0.5
slab_h     = 0.22
design_len = 3.0
design_wid = 2.0
Nx = int(design_len * resolution)
Ny = int(design_wid * resolution)

AUTOSIM_DIR = Path("/root/autosim")
RESULT_DIR  = AUTOSIM_DIR / "results"
RESULT_DIR.mkdir(parents=True, exist_ok=True)

def save_result(pattern_name, outputs=None, error=None, elapsed=0):
    d = RESULT_DIR / pattern_name
    d.mkdir(exist_ok=True)
    data = {
        "pattern": pattern_name,
        "status": "error" if error else "ok",
        "elapsed_s": round(elapsed, 2),
        "outputs": outputs or [],
        "error": str(error) if error else None,
    }
    (d / "result.json").write_text(json.dumps(data, indent=2))
    return data
```

Actually better structure:
```python
#!/usr/bin/env python3
import sys, os, time as _time
sys.path.insert(0, "/root/autosim")
from common import *

_t0 = _time.time()
try:
    # ---- PATTERN CODE BELOW (cleaned) ----
    {cleaned_code}
    # ---- END PATTERN CODE ----
    _outputs = []
    _fig_out = RESULT_DIR / "{name}" / "output.png"
    if plt.get_fignums():
        _fig_out.parent.mkdir(exist_ok=True)
        plt.savefig(str(_fig_out), dpi=100, bbox_inches="tight")
        plt.close("all")
        _outputs.append("output.png")
    save_result("{name}", outputs=_outputs, elapsed=_time.time()-_t0)
except Exception as _e:
    save_result("{name}", error=_e, elapsed=_time.time()-_t0)
    import traceback; traceback.print_exc()
```

Code cleaning rules for {cleaned_code}:
- Remove lines starting with: `import meep`, `import numpy`, `import matplotlib`, `import autograd`, `from meep`, `from numpy`, `from matplotlib`, `from autograd`
- Replace `plt.show()` with `pass  # plt.show() suppressed`
- Replace `plt.savefig(` paths: if relative, prepend `str(RESULT_DIR / "{name}") + "/"`

### Step 4: runner.py
```python
#!/usr/bin/env python3
"""Run all autosim patterns and collect results."""
import subprocess, json, sys, os
from pathlib import Path

PYTHON = "/home/js/miniconda/envs/pmp/bin/python"
PATTERNS_DIR = Path("/root/autosim/patterns")
RESULTS_DIR  = Path("/root/autosim/results")
TIMEOUT = 120  # seconds per pattern (Tier 1/2), use 30 for quick check

def run_pattern(py_file, timeout=TIMEOUT):
    name = py_file.stem
    print(f"  Running: {name}...", flush=True)
    try:
        r = subprocess.run(
            [PYTHON, str(py_file)],
            capture_output=True, text=True,
            timeout=timeout, cwd="/root/autosim"
        )
        result_file = RESULTS_DIR / name / "result.json"
        if result_file.exists():
            return json.loads(result_file.read_text())
        # Fallback
        if r.returncode == 0:
            return {"pattern": name, "status": "ok", "elapsed_s": 0, "outputs": []}
        else:
            return {"pattern": name, "status": "error", "error": r.stderr[-500:]}
    except subprocess.TimeoutExpired:
        return {"pattern": name, "status": "timeout", "error": f"Exceeded {timeout}s"}
    except Exception as e:
        return {"pattern": name, "status": "error", "error": str(e)}

if __name__ == "__main__":
    tier = sys.argv[1] if len(sys.argv) > 1 else "all"
    patterns = sorted(PATTERNS_DIR.glob("*.py"))
    
    # Tier filtering
    TIER1 = ["plot_", "save_", "create_field_animation", "meep_visualization",
             "history_json", "output_directory", "array_metadata"]
    if tier == "1":
        patterns = [p for p in patterns if any(p.name.startswith(t) or t in p.name for t in TIER1)]
    
    results = []
    ok = err = skip = 0
    for pf in patterns:
        r = run_pattern(pf, timeout=60 if tier == "1" else TIMEOUT)
        results.append(r)
        if r["status"] == "ok":
            ok += 1; print(f"    OK ({r.get('elapsed_s',0):.1f}s)")
        elif r["status"] == "timeout":
            skip += 1; print(f"    TIMEOUT")
        else:
            err += 1; print(f"    FAIL: {str(r.get('error',''))[:80]}")
    
    print(f"\n=== Summary: {ok} OK / {err} FAIL / {skip} TIMEOUT / {len(patterns)} total ===")
    Path("/root/autosim/run_summary.json").write_text(json.dumps(results, indent=2))
    print("Saved: /root/autosim/run_summary.json")
```
